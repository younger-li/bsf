    <script>
        // ================== 基础配置 & 本地存储 ==================
        // 系统数据结构（本地存储，永久保存）
        let systemDB = {
            users: [// 初始管理员：手机号13800138000，密码123456
                { phone: "13800138000", pwd: "e10adc3949ba59abbe56e057f20f883e", role: 1, regTime: new Date('2026-01-01').toLocaleString() }
            ],
            rawPools: [],// 原料池
            breedPools: [],// 养殖池
            eggPros: [],// 虫卵产出
            auxMaterials: [],// 辅材
            records: []// 操作记录
        };
        // 全局状态
        let currentLoginUser = { phone: "", role: 0 };// 0=普通用户，1=管理员
        const MONTH_TODAY = new Date().getMonth() + 1;// 当前月份

        // 加载本地数据
        function loadDB() {
            const localData = localStorage.getItem("bsfBreedingSysDB");
            if (localData) systemDB = JSON.parse(localData);
        }
        // 保存数据到本地
        function saveDB() {
            localStorage.setItem("bsfBreedingSysDB", JSON.stringify(systemDB));
        }
        // 新增操作记录
        function addRecord(type, obj, num) {
            systemDB.records.unshift({
                type: type,
                obj: obj,
                num: num,
                time: new Date().toLocaleString()
            });
            // 只保留最新50条记录
            if (systemDB.records.length > 50) systemDB.records.pop();
            saveDB();
        }

        // ================== 通用工具方法 ==================
        // 消息提示
        function showMsg(elId, text, isErr = true) {
            const el = document.getElementById(elId);
            el.innerText = text;
            el.style.color = isErr ? "red" : "green";
            setTimeout(() => el.innerText = "", 2500);
        }
        // MD5加密（简化版，用于密码存储）
        function md5Simple(s) {
            return btoa(unescape(encodeURIComponent(s))).slice(0,32);
        }
        // 获取当前月份的日期
        function isCurrentMonth(dateStr) {
            return new Date(dateStr).getMonth() + 1 === MONTH_TODAY;
        }
        // 打印当前页面
        function printPage() {
            window.print();
        }

        // ================== Excel导出功能 ==================
        // 通用导出方法
        function exportExcel(data, sheetName, fileName) {
            const ws = XLSX.utils.json_to_sheet(data);
            const wb = XLSX.utils.book_new();
            XLSX.utils.book_append_sheet(wb, ws, sheetName);
            XLSX.writeFile(wb, `${fileName}.xlsx`);
        }
        // 导出全部数据
        function exportAllExcel() {
            const wb = XLSX.utils.book_new();
            // 原料池
            XLSX.utils.book_append_sheet(wb, XLSX.utils.json_to_sheet(formatRawForExcel()), "原料池数据");
            // 养殖池
            XLSX.utils.book_append_sheet(wb, XLSX.utils.json_to_sheet(formatBreedForExcel()), "养殖池数据");
            // 虫卵产出
            XLSX.utils.book_append_sheet(wb, XLSX.utils.json_to_sheet(formatEggForExcel()), "虫卵产出数据");
            // 辅材
            XLSX.utils.book_append_sheet(wb, XLSX.utils.json_to_sheet(formatAuxForExcel()), "辅材数据");
            // 操作记录
            XLSX.utils.book_append_sheet(wb, XLSX.utils.json_to_sheet(systemDB.records), "操作记录");
            XLSX.writeFile(wb, `黑水虻养殖系统全量数据_${new Date().toLocaleDateString()}.xlsx`);
            addRecord("系统操作", "导出全量Excel", "全部数据");
        }
        // 格式化各模块数据为Excel格式
        function formatRawForExcel() {
            return systemDB.rawPools.map((item, i) => ({
                序号: i+1,
                原料池名称: item.name,
                原料类型: item.type,
                体积_m³: item.vol,
                当前库存_公斤: item.stock,
                最低库存_公斤: item.min
            }));
        }
        function formatBreedForExcel() {
            return systemDB.breedPools.map((item, i) => ({
                序号: i+1,
                养殖池编号: item.code,
                规格: item.spec,
                负责人: item.user,
                当前幼虫数_只: item.larva,
                状态: item.status
            }));
        }
        function formatEggForExcel() {
            return systemDB.eggPros.map((item, i) => ({
                序号: i+1,
                产出日期: item.date,
                产卵房号: item.room,
                虫卵等级: item.grade,
                产出数量_克: item.num,
                操作员: item.oper
            }));
        }
        function formatAuxForExcel() {
            return systemDB.auxMaterials.map((item, i) => ({
                序号: i+1,
                辅材品名: item.name,
                规格: item.spec,
                单位: item.unit,
                当前库存: item.stock
            }));

        // ================== 用户模块（登录/注册/权限） ==================
        // 注册
        function userRegister() {
            const phone = document.getElementById("loginPhone").value.trim();
            const pwd = document.getElementById("loginPwd").value.trim();
            if (!/^1\d{10}$/.test(phone)) {
                showMsg("loginMsg", "请输入正确的11位手机号");
                return;
            }
            if (pwd.length < 6 || pwd.length > 18) {
                showMsg("loginMsg", "密码长度请保持6-18位");
                return;
            }
            loadDB();
            if (systemDB.users.some(u => u.phone === phone)) {
                showMsg("loginMsg", "该手机号已注册");
                return;
            }
            // 新增用户，默认普通角色
            systemDB.users.push({
                phone: phone,
                pwd: md5Simple(pwd),
                role: 0,
                regTime: new Date().toLocaleString()
            });
            saveDB();
            showMsg("loginMsg", "注册成功，请登录", false);
        }
        // 登录
        function userLogin() {
            const phone = document.getElementById("loginPhone").value.trim();
            const pwd = document.getElementById("loginPwd").value.trim();
            loadDB();
            const user = systemDB.users.find(u => u.phone === phone && u.pwd === md5Simple(pwd));
            if (!user) {
                showMsg("loginMsg", "手机号或密码错误");
                return;
            }
            // 保存当前登录状态
            currentLoginUser = { phone: user.phone, role: user.role };
            document.getElementById("currentUser").innerText = user.phone;
            document.getElementById("currentRole").innerText = user.role === 1 ? "管理员" : "普通用户";
            document.getElementById("currentRole").className = user.role === 1 ? "role-tag" : "role-tag normal";
            // 权限控制：管理员显示用户管理菜单
            if (user.role === 1) {
                document.querySelectorAll(".for-admin").forEach(el => el.classList.remove("hide"));
            } else {
                document.querySelectorAll(".for-admin").forEach(el => el.classList.add("hide"));
            }
            // 切换页面并刷新数据
            document.querySelectorAll(".page").forEach(p => p.classList.remove("active"));
            document.getElementById("mainPage").classList.add("active");
            refreshAllData();
            addRecord("系统操作", "用户登录", user.phone);
        }
        // 退出登录
        function userLogout() {
            if (confirm("确认退出登录吗？")) {

                currentLoginUser = { phone: "", role: 0 };
                document.querySelectorAll(".page").forEach(p => p.classList.remove("active"));
                document.getElementById("loginPage").classList.add("active");
                document.getElementById("loginPhone").value = "";
                document.getElementById("loginPwd").value = "";

        // ================== 首页数据刷新 ==================
        function refreshHomeData() {
            loadDB();
            // 计算核心数据
            const rawStockTotal = systemDB.rawPools.reduce((sum, item) => sum + item.stock, 0);
            const breedNum = systemDB.breedPools.filter(item => item.status === "养殖中").length;
            const eggTotal = systemDB.eggPros.reduce((sum, item) => isCurrentMonth(item.date) ? sum + parseFloat(item.num) : sum, 0);
            const auxNum = systemDB.auxMaterials.length;
            // 更新卡片数据
            document.getElementById("homeRawStock").innerText = rawStockTotal.toFixed(1);
            document.getElementById("homeBreedNum").innerText = breedNum;
            document.getElementById("homeEggTotal").innerText = eggTotal.toFixed(1);
            document.getElementById("homeAuxNum").innerText = auxNum;
            // 库存预警
            const warnList = [];
            systemDB.rawPools.forEach(item => {
                if (item.stock < item.min) warnList.push(`⚠️ 原料池【${item.name}】库存不足（当前：${item.stock.toFixed(1)}公斤，阈值：${item.min}公斤）`);
            });
            systemDB.auxMaterials.forEach(item => {
                if (item.stock < 0) warnList.push(`⚠️ 辅材【${item.name}】库存为负（当前：${item.stock}${item.unit}）`);
            });
            document.getElementById("homeWarnArea").innerHTML = warnList.length > 0 ? `<div class="warn">${warnList.join("<br>")}</div>` : "";
            // 最新操作记录
            const recordTbody = document.getElementById("latestRecordTable").querySelector("tbody");
            recordTbody.innerHTML = "";
            systemDB.records.slice(0,10).forEach(item => {
                recordTbody.innerHTML += `
                <tr>
                    <td>${item.type}</td>
                    <td>${item.obj}</td>
                    <td>${item.num}</td>
                    <td>${item.time}</td>
                </tr>`;
            });
            // 刷新图表
            initHomeCharts();
        }
