2.2.4 报表统计核心接口（ReportController.java）

java

package com.blackfly.management.controller;

import com.blackfly.management.dto.ReportQueryDTO;
import com.blackfly.management.service.ReportService;
import com.blackfly.management.vo.ResultVO;
import com.blackfly.management.vo.ReportVO;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.security.access.prepost.PreAuthorize;
import org.springframework.web.bind.annotation.*;

import javax.servlet.http.HttpServletResponse;
import java.io.IOException;

@RestController
@RequestMapping("/api/report")
public class ReportController {

    @Autowired
    private ReportService reportService;

    // 查看报表（所有用户可访问）
    @PostMapping("/view")
    public ResultVO<ReportVO> viewReport(@RequestBody ReportQueryDTO queryDTO) {
        ReportVO reportVO = reportService.getReport(queryDTO);
        return ResultVO.success(reportVO);
    }

    // 导出报表（仅管理员可操作）
    @PreAuthorize("hasRole('ADMIN')")
    @PostMapping("/export")
    public void exportReport(@RequestBody ReportQueryDTO queryDTO, HttpServletResponse response) throws IOException {
        reportService.exportReport(queryDTO, response);
    }

    // 保存自定义报表模板（仅管理员可操作）
    @PreAuthorize("hasRole('ADMIN')")
    @PostMapping("/saveTemplate")
    public ResultVO<String> saveTemplate(@RequestBody ReportQueryDTO queryDTO) {
        reportService.saveReportTemplate(queryDTO);
        return ResultVO.success("模板保存成功");
    }

    // 批量导出报表（仅管理员可操作）
    @PreAuthorize("hasRole('ADMIN')")
    @PostMapping("/batchExport")
    public void batchExport(@RequestBody Integer[] reportIds, HttpServletResponse response) throws IOException {
        reportService.batchExportReport(reportIds, response);
    }
}

        meta: { requiresAdmin: true }
      }
    ]

const router = createRouter({
  history: createWebHistory(),
  routes
})

// 路由守卫：未登录跳转至登录页，普通用户不可访问管理员页面
router.beforeEach((to, from, next) => {
  const store = useStore()
  const isLogin = store.state.user.isLogin
  const isAdmin = store.state.user.role === 1

  if (to.meta.requiresAuth && !isLogin) {
    next('/login')
  } else if (to.meta.requiresAdmin && !isAdmin) {
    ElMessage.warning('您无此操作权限，请联系管理员')
    next(from.path)
  } else {
    next()
  }
})

export default router
