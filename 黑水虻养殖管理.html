        // MD5加密（简化版，用于密码存储）

        function md5Simple(s) {

            return btoa(unescape(encodeURIComponent(s))).slice(0,32);

        }

        // 获取当前月份的日期

        function exportExcel(data, sheetName, fileName) {

            const ws = XLSX.utils.json_to_sheet(data);

            const wb = XLSX.utils.book_new();

            XLSX.utils.book_append_sheet(wb, ws, sheetName);

            XLSX.writeFile(wb, `${fileName}.xlsx`);

        }

        // ================== 用户模块（登录/注册/权限） ==================

        // 注册

        function userRegister() {

            const phone = document.getElementById("loginPhone").value.trim();

            const pwd = document.getElementById("loginPwd").value.trim();

            if (!/^1\d{10}$/.test(phone)) {

                showMsg("loginMsg", "请输入正确的11位手机号");

                return;

            }

            if (pwd.length < 6 || pwd.length > 18) {

                showMsg("loginMsg", "密码长度请保持6-18位");

                return;

            }

            loadDB();

            if (systemDB.users.some(u => u.phone === phone)) {

                showMsg("loginMsg", "该手机号已注册");

                return;

            }

            // 新增用户，默认普通角色

            systemDB.users.push({

                phone: phone,

                pwd: md5Simple(pwd),

                role: 0,

                regTime: new Date().toLocaleString()

            });

            saveDB();

            showMsg("loginMsg", "注册成功，请登录", false);

        }

        // 登录

        function userLogin() {

            const phone = document.getElementById("loginPhone").value.trim();

            const pwd = document.getElementById("loginPwd").value.trim();

            loadDB();

            const user = systemDB.users.find(u => u.phone === phone && u.pwd === md5Simple(pwd));

            if (!user) {

                showMsg("loginMsg", "手机号或密码错误");

                return;

            }

            // 保存当前登录状态

            currentLoginUser = { phone: user.phone, role: user.role };

            document.getElementById("currentUser").innerText = user.phone;

            document.getElementById("currentRole").innerText = user.role === 1 ? "管理员" : "普通用户";

            document.getElementById("currentRole").className = user.role === 1 ? "role-tag" : "role-tag normal";

            // 权限控制：管理员显示用户管理菜单

            if (user.role === 1) {

                document.querySelectorAll(".for-admin").forEach(el => el.classList.remove("hide"));

            } else {

                document.querySelectorAll(".for-admin").forEach(el => el.classList.add("hide"));

            }

            // 切换页面并刷新数据

            document.querySelectorAll(".page").forEach(p => p.classList.remove("active"));

            document.getElementById("mainPage").classList.add("active");

            refreshAllData();

            addRecord("系统操作", "用户登录", user.phone);

        }

        // 退出登录

        function userLogout() {

            if (confirm("确认退出登录吗？")) {

                localStorage.removeItem("loginUser");

                currentLoginUser = { phone: "", role: 0 };

                document.querySelectorAll(".page").forEach(p => p.classList.remove("active"));

                document.getElementById("loginPage").classList.add("active");

                document.getElementById("loginPhone").value = "";

                document.getElementById("loginPwd").value = "";

                addRecord("系统操作", "用户退出", currentLoginUser.phone);

            }

        }

        // 删除用户（仅管理员，不可删除初始管理员）

        function deleteUser(phone) {

            if (confirm(`确认删除用户【${phone}】吗？`)) {

                loadDB();

                systemDB.users = systemDB.users.filter(u => u.phone !== phone);

                saveDB();

                refreshUser();

                addRecord("系统操作", "删除用户", phone);

            }

        function rawIn(index) {

            const num = prompt("请输入进料数量（公斤）：", 0);

            if (num === null || isNaN(num) || parseFloat(num) <= 0) {

                alert("请输入有效的正数");

                return;

            }

            loadDB();

            systemDB.rawPools[index].stock = (systemDB.rawPools[index].stock || 0) + parseFloat(num);

            saveDB();

            refreshRaw();

            addRecord("进料", systemDB.rawPools[index].name, `${num}公斤`);

        }

        // 原料出料

        function rawOut(index) {

            const num = prompt("请输入出料数量（公斤）：", 0);

            if (num === null || isNaN(num) || parseFloat(num) <= 0) {

                alert("请输入有效的正数");

                return;

            }

            loadDB();

            if (systemDB.rawPools[index].stock < parseFloat(num)) {

                alert("库存不足，无法出料");

                return;

            }

            systemDB.rawPools[index].stock -= parseFloat(num);

            saveDB();

            refreshRaw();

            addRecord("出料", systemDB.rawPools[index].name, `${num}公斤`);

        }

        function auxOut(index) {

            if (num === null || isNaN(num) || parseInt(num) <= 0) {

                alert("请输入有效的正数");

                return;

            }

            loadDB();

            if (systemDB.auxMaterials[index].stock < parseInt(num)) {

                alert("库存不足，无法出库");

                return;

            }

            systemDB.auxMaterials[index].stock -= parseInt(num);

            saveDB();

            refreshAux();

            addRecord("辅材出库", systemDB.auxMaterials[index].name, `${num}${systemDB.auxMaterials[index].unit}`);

        }

 

        // ================== 首页数据刷新 ==================

        function refreshHomeData() {

            loadDB();

            // 计算核心数据

            const rawStockTotal = systemDB.rawPools.reduce((sum, item) => sum + item.stock, 0);

            const breedNum = systemDB.breedPools.filter(item => item.status === "养殖中").length;

            const eggTotal = systemDB.eggPros.reduce((sum, item) => isCurrentMonth(item.date) ? sum + parseFloat(item.num) : sum, 0);

            const auxNum = systemDB.auxMaterials.length;

            // 更新卡片数据

            document.getElementById("homeRawStock").innerText = rawStockTotal.toFixed(1);

            document.getElementById("homeBreedNum").innerText = breedNum;

            document.getElementById("homeEggTotal").innerText = eggTotal.toFixed(1);

            document.getElementById("homeAuxNum").innerText = auxNum;

            // 库存预警

            const warnList = [];

            systemDB.rawPools.forEach(item => {

                if (item.stock < item.min) warnList.push(`⚠️ 原料池【${item.name}】库存不足（当前：${item.stock.toFixed(1)}公斤，阈值：${item.min}公斤）`);

            });

            systemDB.auxMaterials.forEach(item => {

                if (item.stock < 0) warnList.push(`⚠️ 辅材【${item.name}】库存为负（当前：${item.stock}${item.unit}）`);

            });

            document.getElementById("homeWarnArea").innerHTML = warnList.length > 0 ? `<div class="warn">${warnList.join("<br>")}</div>` : "";

            // 最新操作记录

            const recordTbody = document.getElementById("latestRecordTable").querySelector("tbody");

            recordTbody.innerHTML = "";

            systemDB.records.slice(0,10).forEach(item => {

                recordTbody.innerHTML += `

                <tr>

                    <td>${item.type}</td>

                    <td>${item.obj}</td>

                    <td>${item.num}</td>

                    <td>${item.time}</td>

                </tr>`;

            });

            // 刷新图表

            initHomeCharts();

        }

 

        // ================== 全局数据刷新 ==================

        function refreshAllData() {

            refreshRaw();

            refreshBreed();

            refreshEgg();

            refreshAux();

            refreshHomeData();

            if (currentLoginUser.role === 1) refreshUser();

        }
